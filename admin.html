CREATE TABLE cn_profiles ( id UUID DEFAULT gen_random_uuid() PRIMARY KEY, user_id UUID REFERENCES auth.users(id) ON DELETE CASCADE, name TEXT, email TEXT, phone TEXT, height NUMERIC, created_at TIMESTAMPTZ DEFAULT now() );

ALTER TABLE cn_profiles ENABLE ROW LEVEL SECURITY;
CREATE POLICY "admin_all" ON cn_profiles USING (true) WITH CHECK (true);
